### 关键功能实现

#### 1. 主题切换
```jsx
// src/styles/themes.js
export const themes = {
  tech: { primary: '#00D4FF', bg: '#0A192F' },
  warm: { primary: '#FF8C42', bg: '#FFF5F0' },
  simple: { primary: '#000000', bg: '#FFFFFF' },
  active: { primary: '#7FB069', bg: '#F0FFF4' }
};
```

#### 2. 网络图片获取
```jsx
// src/utils/api.js
export async function fetchImages(keyword) {
  const res = await axios.get(
    `https://api.unsplash.com/search/photos?query=${keyword}`,
    { headers: { Authorization: `Bearer ${ACCESS_KEY}` }}
  );
  return res.data.results;
}
```

#### 3. 丝滑动画
```jsx
// 使用 Framer Motion
<motion.div
  initial={{ opacity: 0, y: 30 }}
  animate={{ opacity: 1, y: 0 }}
  transition={{ duration: 0.5, ease: "easeOut" }}
>
  内容
</motion.div>
```
